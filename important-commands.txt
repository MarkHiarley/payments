# 1. REVERTER A ÚLTIMA MIGRATION
migrate -path ./internal/database/migrations -database "$DB_URL" down 1

# 2. REVERTER TODAS AS MIGRATIONS
migrate -path ./internal/database/migrations -database "$DB_URL" down

# 3. REVERTER MÚLTIPLAS MIGRATIONS (exemplo: 2 últimas)
migrate -path ./internal/database/migrations -database "$DB_URL" down 2

# 4. IR PARA UMA VERSÃO ESPECÍFICA
migrate -path ./internal/database/migrations -database "$DB_URL" goto 1

# 5. FORÇAR UMA VERSÃO (útil quando há dirty state)
migrate -path ./internal/database/migrations -database "$DB_URL" force 2
